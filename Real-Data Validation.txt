import warnings
warnings.filterwarnings("ignore")

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import networkx as nx
import yfinance as yf

from GraphRicciCurvature.OllivierRicci import OllivierRicci

# ============================================================
# REAL-DATA GEOMETRIC-SPECTRAL VALIDATION
# ------------------------------------------------------------
# Data: 11 U.S. sector ETFs
# Period: 2018-01-01 to 2021-12-31
# Goal:
#   1. Download real price data
#   2. Compute rolling log-return correlation matrices
#   3. Build regularized distance matrix:
#          d_ij = -log((1 + C_ij)/2)
#   4. Build topology-preserving k-NN graph
#   5. Compute average Ollivier-Ricci curvature
#   6. Compute leading eigenvalue lambda_1
#   7. Compute Marchenko-Pastur upper edge lambda_+
#   8. Plot:
#        - lambda_1(t) vs lambda_+
#        - average curvature kappa(t)
#        - phase plot kappa(t) vs lambda_1(t)
# ============================================================

# -----------------------------
# 1. Configuration
# -----------------------------
TICKERS = [
    "XLB",  # Materials
    "XLE",  # Energy
    "XLF",  # Financials
    "XLI",  # Industrials
    "XLK",  # Technology
    "XLP",  # Consumer Staples
    "XLU",  # Utilities
    "XLV",  # Health Care
    "XLY",  # Consumer Discretionary
    "XLRE", # Real Estate
    "XLC",  # Communication Services
]

START_DATE = "2018-01-01"
END_DATE   = "2021-12-31"

ROLLING_WINDOW = 126      # about 6 months of trading days
STEP = 5                  # weekly update
K_NEIGHBORS = 4           # k-NN graph
RICCI_ALPHA = 0.5         # standard parameter in OllivierRicci
EPS = 1e-8                # numerical floor for distance regularization

# -----------------------------
# 2. Download real data
# -----------------------------
# Explicit auto_adjust=True for robust total-return-like close series.
raw = yf.download(
    TICKERS,
    start=START_DATE,
    end=END_DATE,
    auto_adjust=True,
    progress=False,
    threads=True
)

if raw.empty:
    raise ValueError("No data returned by yfinance. Check internet/ticker availability.")

# yfinance returns multi-index columns for multiple tickers.
# Use Close if available; otherwise fallback.
if isinstance(raw.columns, pd.MultiIndex):
    if "Close" in raw.columns.get_level_values(0):
        prices = raw["Close"].copy()
    else:
        # fallback: first level if Close not present
        prices = raw.xs(raw.columns.levels[0][0], axis=1, level=0).copy()
else:
    prices = raw.copy()

prices = prices.dropna(how="all").ffill().dropna()

# -----------------------------
# 3. Compute log returns
# -----------------------------
log_returns = np.log(prices / prices.shift(1)).dropna()

# Ensure enough rows
if len(log_returns) <= ROLLING_WINDOW:
    raise ValueError("Not enough observations for the chosen rolling window.")

# -----------------------------
# 4. Helper functions
# -----------------------------
def regularized_distance_from_correlation(C: pd.DataFrame, eps: float = 1e-8) -> pd.DataFrame:
    """
    Compute regularized distance matrix:
        d_ij = -log((1 + C_ij)/2)
    with clipping for numerical safety.
    """
    C_clipped = C.clip(lower=-1 + eps, upper=1 - eps)
    D = -np.log((1.0 + C_clipped) / 2.0)
    np.fill_diagonal(D.values, 0.0)
    return D


def build_knn_graph(D: pd.DataFrame, k: int) -> nx.Graph:
    """
    Build a symmetric k-NN graph from a distance matrix.
    Edge weight is set to the distance.
    """
    nodes = list(D.index)
    G = nx.Graph()
    G.add_nodes_from(nodes)

    for i in nodes:
        # sort neighbors by increasing distance, exclude self
        neighbors = D.loc[i].drop(index=i).sort_values().head(k).index.tolist()
        for j in neighbors:
            w = float(D.loc[i, j])
            if np.isfinite(w):
                G.add_edge(i, j, weight=w)

    # Ensure graph is connected if possible:
    # If not connected, connect components using shortest inter-component edges.
    if not nx.is_connected(G):
        components = list(nx.connected_components(G))
        while len(components) > 1:
            comp_a = list(components[0])
            comp_b = list(components[1])

            best_edge = None
            best_dist = np.inf
            for a in comp_a:
                for b in comp_b:
                    dist_ab = float(D.loc[a, b])
                    if dist_ab < best_dist:
                        best_dist = dist_ab
                        best_edge = (a, b)

            G.add_edge(best_edge[0], best_edge[1], weight=best_dist)
            components = list(nx.connected_components(G))

    return G


def average_ollivier_ricci_curvature(G: nx.Graph, alpha: float = 0.5) -> float:
    """
    Compute average Ollivier-Ricci curvature over graph edges.
    """
    orc = OllivierRicci(G, alpha=alpha, verbose="ERROR")
    orc.compute_ricci_curvature()
    Gc = orc.G

    curvatures = []
    for u, v, data in Gc.edges(data=True):
        if "ricciCurvature" in data:
            curvatures.append(float(data["ricciCurvature"]))

    if len(curvatures) == 0:
        return np.nan
    return float(np.mean(curvatures))


def leading_eigenvalue(C: pd.DataFrame) -> float:
    """
    Largest eigenvalue of symmetric correlation matrix.
    """
    vals = np.linalg.eigvalsh(C.values)
    return float(vals[-1])


def mp_upper_edge(n_assets: int, window_length: int) -> float:
    """
    Marchenko-Pastur upper edge:
        lambda_+ = (1 + sqrt(1/Q))^2,  Q = T / N >= 1
    """
    Q = window_length / n_assets
    if Q < 1:
        raise ValueError("Need Q = T/N >= 1 for the MP formula used here.")
    return float((1.0 + np.sqrt(1.0 / Q)) ** 2)


# -----------------------------
# 5. Rolling computation
# -----------------------------
results = []

n_assets = log_returns.shape[1]
lambda_plus = mp_upper_edge(n_assets=n_assets, window_length=ROLLING_WINDOW)

dates = log_returns.index

for end_idx in range(ROLLING_WINDOW, len(log_returns), STEP):
    window = log_returns.iloc[end_idx - ROLLING_WINDOW:end_idx].copy()
    date_t = dates[end_idx]

    # Correlation matrix
    C = window.corr()

    # Leading eigenvalue
    lam1 = leading_eigenvalue(C)

    # Regularized distance
    D = regularized_distance_from_correlation(C, eps=EPS)

    # k-NN topology-preserving graph
    G = build_knn_graph(D, k=K_NEIGHBORS)

    # Average Ollivier-Ricci curvature
    kappa_avg = average_ollivier_ricci_curvature(G, alpha=RICCI_ALPHA)

    results.append({
        "Date": date_t,
        "lambda_1": lam1,
        "kappa_avg": kappa_avg,
        "lambda_plus": lambda_plus,
        "num_nodes": G.number_of_nodes(),
        "num_edges": G.number_of_edges(),
    })

results_df = pd.DataFrame(results).set_index("Date")

# -----------------------------
# 6. Crisis flag (for visualization)
# -----------------------------
# A simple instability indicator:
#   spectral outlier + low curvature relative to rolling baseline
results_df["spectral_outlier"] = results_df["lambda_1"] > results_df["lambda_plus"]

# Curvature threshold: 20th percentile over the whole sample
kappa_threshold = results_df["kappa_avg"].quantile(0.20)
results_df["low_curvature"] = results_df["kappa_avg"] <= kappa_threshold

results_df["instability_flag"] = (
    results_df["spectral_outlier"] & results_df["low_curvature"]
)

# -----------------------------
# 7. Print summary
# -----------------------------
print("=" * 70)
print("REAL-DATA GEOMETRIC-SPECTRAL VALIDATION")
print("=" * 70)
print(f"Assets: {n_assets}")
print(f"Rolling window: {ROLLING_WINDOW}")
print(f"Step: {STEP}")
print(f"MP upper edge lambda_+: {lambda_plus:.4f}")
print(f"Curvature threshold (20th percentile): {kappa_threshold:.4f}")
print()
print("First rows:")
print(results_df.head())
print()
print("Rows flagged as instability:")
print(results_df[results_df["instability_flag"]].head(20))

# -----------------------------
# 8. Plot 1: lambda_1(t) vs MP bound
# -----------------------------
plt.figure(figsize=(12, 5))
plt.plot(results_df.index, results_df["lambda_1"], label=r"$\lambda_1(t)$")
plt.axhline(results_df["lambda_plus"].iloc[0], linestyle="--", label=r"$\lambda_+$ (MP upper edge)")
plt.title("Leading Eigenvalue vs Marchenko-Pastur Upper Edge")
plt.xlabel("Date")
plt.ylabel("Eigenvalue")
plt.legend()
plt.tight_layout()
plt.show()

# -----------------------------
# 9. Plot 2: average curvature over time
# -----------------------------
plt.figure(figsize=(12, 5))
plt.plot(results_df.index, results_df["kappa_avg"], label=r"$\langle \kappa \rangle(t)$")
plt.axhline(kappa_threshold, linestyle="--", label="Curvature threshold")
plt.title("Average Ollivier-Ricci Curvature Over Time")
plt.xlabel("Date")
plt.ylabel("Average curvature")
plt.legend()
plt.tight_layout()
plt.show()

# -----------------------------
# 10. Plot 3: phase diagram
# -----------------------------
plt.figure(figsize=(7, 6))
plt.scatter(results_df["kappa_avg"], results_df["lambda_1"], s=20)
plt.axhline(results_df["lambda_plus"].iloc[0], linestyle="--", label=r"$\lambda_+$")
plt.axvline(kappa_threshold, linestyle="--", label=r"$\kappa_c$")
plt.title("Geometric-Spectral Phase Diagram")
plt.xlabel(r"Average curvature $\langle \kappa \rangle(t)$")
plt.ylabel(r"Leading eigenvalue $\lambda_1(t)$")
plt.legend()
plt.tight_layout()
plt.show()

# -----------------------------
# 11. Optional: save results
# -----------------------------
results_df.to_csv("real_data_geometric_spectral_results.csv")
print("\nSaved results to: real_data_geometric_spectral_results.csv")
